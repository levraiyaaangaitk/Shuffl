SMODS.Joker{
    key = 'dodgeroll',
    config = {
        extra = {
            outof = 15
        }
    },
    loc_txt = {
        name = 'Dodgeroll',
        text = {
            '{C:green}#1# in #2#{} chance to',
            'make blind requirement {C:red}25%{} lower',
            'when the blind is selected.'
        }
    },
    rarity = 3,
    cost = 6,
    blueprint_compat=true,
    eternal_compat=true,
    unlocked = true,
    discovered = true,
    loc_vars = function(self, info_queue, card)
        return { vars = { G.GAME.probabilities.normal, card.ability.extra.outof } }
    end,
    calculate = function (self,card,context)
        if context.before then
            if 1 == 1 then
                G.GAME.blind.chips = G.GAME.blind.chips * 0.75
                return{
                    card = card,
                    colour = G.C.RED,
                    message = 'Dodged !'
                }
            end
        end
    end
}
